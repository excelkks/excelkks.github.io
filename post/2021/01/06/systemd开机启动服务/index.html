2021-01-06

systemd开机运行服务

[Unit]
Description=renew ipv6 (ddns)
After=network.target # needed service

[Service]
Type=oneshot  # start once
ExecStart=/bin/bash /etc/scripts  # what to execute

[Install]
WantedBy=multi-user.target  # startup execute
